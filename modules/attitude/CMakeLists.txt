file(GLOB_RECURSE INC "*.h")
file(GLOB_RECURSE SRC "*.c" "*.cpp")

include_directories(${LIB_DIR}/sdk/include ${CMAKE_CURRENT_BINARY_DIR})
include_directories(${RFS_DIR}/usr/local/include/libbson-1.0)
include_directories(${CMAKE_CURRENT_BINARY_DIR}/../../libraries/sdk)

set(EXTRA_LIBS
    sdk
    m
    pthread
    rt
)

ADD_CUSTOM_COMMAND(
    OUTPUT ${CMAKE_CURRENT_BINARY_DIR}/attitude.helper.c
    COMMAND ${NODEJS} ${LIB_DIR}/sdk/ModuleGenerator.js ${CMAKE_CURRENT_SOURCE_DIR}/attitude.def.json ${CMAKE_CURRENT_BINARY_DIR} ${PLATFORM}
    DEPENDS ${CMAKE_CURRENT_SOURCE_DIR}/attitude.def.json
)

add_executable(attitude ${INC} ${SRC} attitude.helper.c)

target_link_libraries(attitude ${EXTRA_LIBS})

IF(${DO_POST_BUILD})
    add_custom_command(
        TARGET attitude
        POST_BUILD
        COMMAND ${SCP} -P ${SSH_PORT_TARGET_SYSTEM} attitude ${URL_TARGET_SYSTEM}:/usr/local/linuxdrone/bin/attitude.mod
    )
ENDIF()

install(TARGETS attitude DESTINATION bin)
